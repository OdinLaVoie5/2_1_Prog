mkdir hexlet-git    создать папку
	$ mkdir -p dir1/dir2/dir3   создать папку в папке в папке 
	$ touch file.txt  -создать файл(файлы через пробел)   
	можно использовать обе команды вместе с символом домашней директории (`~`) или родительской директории (`..`)
	$ touch ../../file.txt   создать файл `file.txt` на две папки выше по иерархии.
cd hexlet-git                      перейти в неё (hexlet-git можно заменить)
дополнительно 
	( pwd - где я?      cd ~   - в домашнюю директорию  (user)       cd ..   - переходим на уровень выше    ls  - отобразить содержимое директории  (с флагом `ls -a` и вывести расширенный список)  `ls ~`  -содержимое домашней директории    `ls ..` -содержимое родительской директории)
	$ cp что_копируем куда_копируем
	$ cp что_копируем что_копируем что_копируем куда_копируем
	$ mv что_перемещаем куда_перемещаем
	$ mv что_перемещаем что_перемещаем куда_перемещаем 
	$ cat text.txt # распечатать содержимое файла (только с текстовыми файлами)
	$ rm example.txt   удалить файл        $ rmdir dir1 удалить папку 
	$ rm -r dir1      удалить папку со всем её содержимым
