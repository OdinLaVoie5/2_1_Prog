function benchmarkMyMy (num, numOut, argum, func,  func1, func2) {   
  function bench(f) {                    // ф-ция замера разницы
	var date = Date.now();        //performance.now(); - вариант более точно для брауза.
	for (var i = 0; i < num; i++) f(argum);  // num раз тестируемая ф-ция
	return Date.now() - date;     //  разница.  performance.now() - date; - вариант 
  }
  // bench для каждого теста запустим много раз, чередуя
  var time = 0, time1 = 0, time2 = 0;
  // numOut раз  ф-ция замера разницы bench поочерёдно для ф-ция 1 и ф-ция 2
  for (var i = 0; i < numOut; i++) {  
	if(func) time += bench(func);
	if(func1) time1 += bench(func1);
	if(func2) time2 += bench(func2);
  }
  return [time, time1, time2];
}

		if(lastS > lastF){        // не закрыт нач ком
			if(lastS == arrStr[i].length - 3){arrStr[i] = arrStr[i].slice(0, lastS) + '\n' + '/*'+ '\n'; }  // последний в строке
			else {arrStr[i] = arrStr[i].slice(0, lastS) + '\n' + '/*'+ '\n' + arrStr[i].slice(lastS+2); }
		}
		if(indS > indF){         // не закрыт кон ком
			if(indF == 0){arrStr[i] = '*/' + '\n' + arrStr[i].slice(indF+2);}   // первый в строке
			else {arrStr[i] = arrStr[i].slice(0 , indF) + '\n' + '*/' + '\n' + arrStr[i].slice(indF+2);}
		}
	  }  
	  else {     //  в строке остались только закрытые больш коменты 
		// ?????   перенос перед и после,  в начале ставить //    (а если */ /* ?)
		arrStr[i] = arrStr[i].split('/*');
		arrStr[i] = arrStr[i].join('/*' + '<mark style="background-color:Blue">');
		arrStr[i] = arrStr[i].split('*/');
		arrStr[i] = arrStr[i].join('*/' + '</mark>'); // ??  порно когда возле / * * /   //

	  };
	}else{                                     // ЕСТЬ ОДИН ЗНАК  
	  if(~indS){                          // есть  / *
		  arrStr[i] = arrStr[i].split('\/*');
		  if(arrStr[i][0]==0 && arrStr[i][1]!=0) {arrStr[i] = arrStr[i].join('/*'+'\n');}
		  else if(arrStr[i][0]!= 0 && arrStr[i][1]!= 0) {arrStr[i] = arrStr[i].join('\n'+'/*'+'\n');}
		  else if(arrStr[i][0]!= 0 && arrStr[i][1]== 0) {arrStr[i] = arrStr[i].join('\n'+'/*');}	
		  else  {arrStr[i] = arrStr[i].join('/*');}
	  }
	  if(~indF){                          // есть  * /
		  arrStr[i] = arrStr[i].split('*\/');
		  if(arrStr[i][0]==0 && arrStr[i][1]!=0) {arrStr[i] = arrStr[i].join('*/'+'\n');}
		  else if(arrStr[i][0]!=0 && arrStr[i][1]!=0) {arrStr[i] = arrStr[i].join('\n'+'*/'+'\n');}
		  else if(arrStr[i][0]!=0 && arrStr[i][1]==0) {arrStr[i] = arrStr[i].join('\n'+'*/');}
		  else  {arrStr[i] = arrStr[i].join('*/');}
	  }
	}
  } 
  return arrStr.join('');
}
